Phase 3 Stretch Goal: WebGPU Compute

⚡ Phase 3 Preview

WebGPU compute shaders are the natural extension of Selvr's auto-targeting architecture. Just as the compiler routes float-heavy functions to WASM today, a future pass will route massively parallel workloads (matrix operations, image convolutions, physics simulations) to the GPU using WebGPU compute pipelines — with zero developer involvement.

// selvr source — developer writes one function, compiler decides where it runs #[gpu] // Phase 3: explicit override OR auto-detected by targeting pass export fn matrix_multiply(A: f32[], B: f32[], n: i32): f32[] { // Dense f32 loop over n² elements → targeting score >> 200 → GPU route let C: f32[] = []; let i: i32 = 0; while i < n * n { C.push(0.0); i = i + 1; } let row: i32 = 0; while row < n { let col: i32 = 0; while col < n { let s: f32 = 0.0; let k: i32 = 0; while k < n { s = s + A[row*n+k] * B[k*n+col]; k = k+1; } C[row*n+col] = s; col = col + 1; } row = row + 1; } return C; }
